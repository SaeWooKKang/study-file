## 별 그리기 
- 뭔가에 해당되는 일을 여러번 반복할 때 시작을 range로 할 수 있음 

``` javascript

log([..._.range(1, 6)]); // [1, 2, 3, 4, 5]

_.go(
  _.range(1, 6), // [1, 2, 3, 4, 5]
  _.map(_.range), // [[1], [1, 2], [1, 2, 3]...]
  _.map(_.map(_ => '*')), // [['*'], ['*', '*'],['*','*','*']...]
  _.map(_.reduce((a, b) => `${a}${b}`)),// ['*', '**', '***', '****', '*****']
  _.reduce((a, b) => `${a}\n${b}`),
  log
);

log(_.map(_.range, [1, 2, 3])); // [[1], [1, 2], [1, 2, 3]]

// _.map(f, list)
log(_.map(_.map(_=> '*'), [[1], [1, 2], [1, 2, 3]]));
log(_.map(_ => '*', [1,2])); // 배열 길이만큼 배열에 * 담아 반환하는구나

const join = sep => _.reduce((a, b) => `${a}${sep}${b}`);

_.go(
  L.range(1, 6), // [1, 2, 3, 4, 5]
  L.map(L.range), // [[1], [1, 2], [1, 2, 3]...]
  L.map(L.map(_ => '*')), // [['*'], ['*', '*'],['*','*','*']...]
  L.map(join('')),// ['*', '**', '***', '****', '*****']
  join(`\n`),
  log
);
```

## 구구단
``` javascript
_.go(
  _.range(2,10), // [2, 3, 4, 5, 6, 7, 8, 9]
  _.map(a => _.go(
    _.range(1, 10), // [1, 2, 3, 4, 5, 6, 7, 8, 9]
    _.map(b => `${a}x${b}=${a*b}`),
    join(`\n`),
  )),
  join(`\n\n`),
  log
);
```
